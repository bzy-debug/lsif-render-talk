53///   inspect!(2.0.sinh(), content="3.626860407847019")
54///   inspect!(1000.0.sinh(), content="Infinity")
55///   inspect!((-1000.0).sinh(), content="-Infinity")
56///   inspect!(@double.not_a_number.sinh(), content="NaN")
57///   inspect!(@double.infinity.sinh(), content="Infinity")
58///   inspect!(@double.neg_infinity.sinh(), content="-Infinity")
