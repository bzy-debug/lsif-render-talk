  inspect!((-1.0).atanh(), content="-Infinity")
  inspect!(1.5.atanh(), content="NaN")
  inspect!(@double.not_a_number.atanh(), content="NaN")
  inspect!(@double.infinity.atanh(), content="NaN")
  inspect!(@double.neg_infinity.atanh(), content="NaN")
